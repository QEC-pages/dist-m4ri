91static inline int std_bitcount ( uint64_t x) {
92 uint64_t c1 = UINT64_C (0x5555555555555555 );
93 uint64_t c2 = UINT64_C (0x3333333333333333 );
94 uint64_t c4 = UINT64_C (0x0F0F0F0F0F0F0F0F );
95 x -= (x >> 1) & c1;
96 x = (( x >> 2) & c2) + (x & c2);

153 static inline int nextelement(const word * const set1, const int m, const int pos){
154 word setwd;
155 int w;
156 if (pos < 0){
157 w = 0;
158 setwd = set1[0];
159 }
160 else{
161 w = SETWD(pos);
162 setwd = set1[w] & (m4ri_ffff<< SETBT(pos));
163 }
164
165 for (;;){
166 if (setwd != 0)
167 return TIMESWORDSIZE(w) + FIRSTBIT(setwd);
168 if (++w == m)
169 return -1;
170 setwd = set1[w];
171 }
172 }

365static inline int sparse_syndrome_non_zero(const csr_t * const H, const int cnt, const int ee[]){
366 for(int ir=0; ir < H->rows; ir++){
367 int nz=0;

369 int ic = H->i[iL];
