158mzd_t *mzd_from_csr(mzd_t *dst, const csr_t *p) {
159 int m=p->rows, n=p->cols, nz=p->nz;
160 if (dst == NULL)
161 dst = mzd_init(m,n);
162 else if ((dst->nrows != m) || (dst->ncols != n))
163 ERROR("Wrong size for return matrix.\n");
164 else
165 mzd_set_ui(dst,0); /* clear bits */
166
167 if(nz==-1){ // binary CSR matrix in Compressed Row Form
168 for(int i=0;i<m;i++)
169 for(int j=p->p[i]; j < p->p[i+1] ; j++)
170 mzd_write_bit(dst, i, p->i[j], 1);
171 }
172 else{ // binary CSR matrix in Pair Form
173 for(int i=0;i<p->nz;i++)
174 mzd_write_bit(dst, p->p[i],p->i[i],1);
175 }

514void csr_compress(csr_t *mat){
515 int nz=mat->nz;
516 if(nz==-1)
517 ERROR("matrix already compressed");
518 int_pair *pairs=calloc(nz,sizeof(int_pair));
519 for(int i=0;i<nz;i++){
520 pairs[i].a=mat->p[i];
521 pairs[i].b=mat->i[i];
522 }
523 qsort(pairs,nz,sizeof(int_pair),cmp_int_pairs);
524 int i, j=0;
525 for(i=0;i<mat->rows;i++){
526 mat->p[i]=j;
527 while ((j<nz)&&(pairs[j].a == i)){
528 mat->i[j]=pairs[j].b;
529 j++;
530 }
531 }
532 mat->p[i]=j; /* final value */
533 mat->nz=-1; /* indicate compressed form */
534 free(pairs);
535}

770csr_t * Lx_for_CSS_code(const csr_t * const Hx, const csr_t *const Hz){
771 rci_t n=Hx->cols;
772 if(n!=Hz->cols)
773 ERROR("unequal number of columns in matrices Hx[%d,%d] and Hz[%d,%d]",
774 Hx->rows, Hx->cols, Hz->rows, Hz->cols);
775 mzd_t *Mx = mzd_from_csr(NULL,Hx); /* convert to dense matrix */
776 mzp_t *pivots = mzp_init(n); /* initialize the permutation */
777 rci_t rank = mzd_gauss_naive(Mx, pivots, 1);
778 mzd_apply_p_right_trans(Mx,pivots);
779 mzp_t *perm = perm_p( NULL, pivots,0);
780 mzp_t *permT = perm_p_trans(NULL, pivots,0);
781 mzp_free(pivots);
782 csr_t* HzPerm=csr_apply_perm(NULL,Hz,permT);
783 mzp_free(permT);
784#ifndef NDEBUG
785 mzd_t *MxT = mzd_transpose(NULL,Mx);
786 if(product_weight_csr_mzd(HzPerm,MxT,0))
787 ERROR("rows of Hx and Hz should be orthogonal ");
788 mzd_free(MxT);
789#endif
790 mzd_t *MzStar = mzd_generator_from_csr(NULL, HzPerm);
791 csr_free(HzPerm);
792
793 mzd_t *mat = mzd_stack(NULL,Mx,MzStar);
794 mzd_free(MzStar);
795 // printf("stacked:\n"); mzd_print(mat);
796 int rank1=mzd_echelonize(mat,0);
797 // printf("after gauss %d -> %d:\n", rank, rank1); mzd_print(mat);
798
799 mzd_t *window = mzd_init_window(mat,rank,0,rank1,n);
800 csr_t *Lx=csr_from_mzd(NULL, window);
801 csr_t *ans = csr_apply_perm(NULL,Lx,perm);
802 mzd_free(Mx);
803 csr_free(Lx);
804 mzd_free(window);
805 mzp_free(perm);
806 mzd_free(mat);
